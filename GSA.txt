const SPREADSHEET_ID = '1ggy2XD0i825rMEf0saZEALcb2mgyYRXJfQKcZz9mHmI';
const HEADERS = ['타입','매장','메뉴','상세메뉴','날짜','시간','성인','아동','유아','예약자','연락처'];
const SETTINGS_SHEET = 'Settings';

function getReservationSheet() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheets = ss.getSheets();
  return sheets.find(s => s.getName() !== SETTINGS_SHEET) || sheets[0];
}

function ensureHeaders(sheet) {
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(HEADERS);
    sheet.getRange(1, 1, 1, HEADERS.length)
         .setFontWeight('bold')
         .setBackground('#f3f3f3');
  }
}

function digitsOnly(phone) {
  return String(phone).replace(/[^0-9]/g, '');
}

function formatSheetValue(value, header) {
  if (!(value instanceof Date)) return value;
  const h  = String(value.getHours()).padStart(2, '0');
  const mi = String(value.getMinutes()).padStart(2, '0');
  if (value.getFullYear() <= 1900) return h + ':' + mi;
  const y  = value.getFullYear();
  const mo = value.getMonth() + 1;
  const d  = value.getDate();
  if (header === '날짜' || (h === '00' && mi === '00')) return y + '년 ' + mo + '월 ' + d + '일';
  return y + '. ' + mo + '. ' + d + '. ' + h + ':' + mi;
}

function doGet(e) {
  const action = (e.parameter && e.parameter.action) || '';
  if (action === 'check')  return handleCheck(e);
  if (action === 'cancel') return handleCancel(e);
  return getSettings();
}

function getSettings() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(SETTINGS_SHEET);
  if (!sheet) {
    sheet = ss.insertSheet(SETTINGS_SHEET);
    sheet.appendRow(['key', 'value']);
    [
      ['booking_days',   '7'],
      ['blocked_dates',  '{}'],
      ['notice_enabled', 'false'],
      ['notice_text',    '']
    ].forEach(r => sheet.appendRow(r));
  }
  const rows = sheet.getDataRange().getValues();
  const settings = {};
  for (let i = 1; i < rows.length; i++) {
    if (rows[i][0]) settings[rows[i][0]] = rows[i][1];
  }
  return ContentService.createTextOutput(JSON.stringify(settings))
    .setMimeType(ContentService.MimeType.JSON);
}

function saveSettings(settingsData) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(SETTINGS_SHEET);
  if (!sheet) {
    sheet = ss.insertSheet(SETTINGS_SHEET);
    sheet.appendRow(['key', 'value']);
  }
  const rows = sheet.getDataRange().getValues();
  for (const [key, value] of Object.entries(settingsData)) {
    let found = false;
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] === key) {
        sheet.getRange(i + 1, 2).setValue(value);
        found = true;
        break;
      }
    }
    if (!found) sheet.appendRow([key, value]);
  }
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);

    if (data.action === 'saveSettings') {
      saveSettings(data.settings);
      return ContentService.createTextOutput(JSON.stringify({ success: true }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    const sheet = getReservationSheet();
    ensureHeaders(sheet);

    sheet.appendRow([
      '예약',
      data.store, data.menu, data.menuDetail || '', data.date, data.time,
      data.adult || 0, data.child || 0, data.infant || 0,
      data.name, String(data.phone)
    ]);

    const lastRow     = sheet.getLastRow();
    const phoneColNum = HEADERS.indexOf('연락처') + 1;
    const phoneStr    = String(data.phone).replace(/['"]/g, '');
    sheet.getRange(lastRow, phoneColNum)
         .setNumberFormat('@')
         .setFormula('="' + phoneStr + '"');
    const menuColNum = HEADERS.indexOf('메뉴') + 1;
    sheet.getRange(lastRow, menuColNum).setNumberFormat('@').setValue(data.menu);

    return ContentService.createTextOutput(JSON.stringify({ success: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleCheck(e) {
  const output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);
  const searchPhone = digitsOnly(e.parameter.phone || '');
  if (searchPhone.length < 9) {
    output.setContent(JSON.stringify({ error: '올바른 연락처를 입력해 주세요.' }));
    return output;
  }
  const sheet = getReservationSheet();
  const data  = sheet.getDataRange().getValues();
  if (data.length < 2) { output.setContent(JSON.stringify({ results: [] })); return output; }

  const headers  = data[0].map(h => String(h).trim());
  const phoneIdx = headers.indexOf('연락처');
  if (phoneIdx === -1) {
    output.setContent(JSON.stringify({ error: '시트에서 연락처 컬럼을 찾을 수 없습니다.' }));
    return output;
  }

  const results = [];
  for (let i = 1; i < data.length; i++) {
    if (digitsOnly(data[i][phoneIdx]) === searchPhone) {
      const row = { _rowIndex: i + 1 };
      headers.forEach((h, idx) => { row[h] = formatSheetValue(data[i][idx], h); });
      row.store      = row['매장'];
      row.menu  = row['상세메뉴'] || row['메뉴'];
      row.date   = row['날짜'];  row.time   = row['시간'];
      row.adult  = row['성인'];  row.child  = row['아동'];
      row.infant = row['유아'];  row.name   = row['예약자'];
      row.phone  = row['연락처'];
      row.status = row['타입'];
      results.push(row);
    }
  }
  output.setContent(JSON.stringify({ results }));
  return output;
}

function handleCancel(e) {
  const output = ContentService.createTextOutput();
  output.setMimeType(ContentService.MimeType.JSON);
  try {
    const sheet      = getReservationSheet();
    const rowIndex   = parseInt(e.parameter.rowIndex);
    const inputPhone = digitsOnly(e.parameter.phone || '');
    const headers    = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(h => String(h).trim());
    const rowData    = sheet.getRange(rowIndex, 1, 1, sheet.getLastColumn()).getValues()[0];
    const phoneIdx   = headers.indexOf('연락처');
    const typeIdx    = headers.indexOf('타입');

    if (digitsOnly(rowData[phoneIdx]) !== inputPhone) {
      output.setContent(JSON.stringify({ success: false, error: '취소 권한이 없습니다.' }));
      return output;
    }

    sheet.getRange(rowIndex, 1, 1, sheet.getLastColumn()).setBackground('#ffcccc');
    if (typeIdx !== -1) sheet.getRange(rowIndex, typeIdx + 1).setValue('취소');

    output.setContent(JSON.stringify({ success: true }));
    return output;
  } catch (err) {
    output.setContent(JSON.stringify({ success: false, error: err.message }));
    return output;
  }
}
